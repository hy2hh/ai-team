## `src/utils/ReconnectingWebSocket.ts`

```typescript
/** WebSocket 연결 상태 */
export type WebSocketStatus =
  | 'connecting'
  | 'connected'
  | 'disconnected'
  | 'reconnecting'
  | 'failed';

/** ReconnectingWebSocket 생성 옵션 */
export interface ReconnectingWebSocketOptions {
  /** 최초 재연결 대기 시간 (ms), 기본값 1000 */
  initialDelay?: number;
  /** 최대 재연결 대기 시간 (ms), 기본값 30000 */
  maxDelay?: number;
  /** 최대 재연결 시도 횟수 (0 = 무제한), 기본값 10 */
  maxRetries?: number;
  /** 지수 백오프 배율, 기본값 2 */
  backoffMultiplier?: number;
  /** 연결 상태 변경 콜백 */
  onStatusChange?: (status: WebSocketStatus) => void;
}

/** ReconnectingWebSocket 이벤트 핸들러 맵 */
export interface WebSocketEventHandlers {
  onOpen?: (event: Event) => void;
  onMessage?: (event: MessageEvent) => void;
  onClose?: (event: CloseEvent) => void;
  onError?: (event: Event) => void;
}

const DEFAULT_OPTIONS: Required<
  Omit<ReconnectingWebSocketOptions, 'onStatusChange'>
> = {
  initialDelay: 1000,
  maxDelay: 30000,
  maxRetries: 10,
  backoffMultiplier: 2,
};

/**
 * 자동 재연결을 지원하는 WebSocket 래퍼 클래스.
 * 지수 백오프(exponential backoff) 전략으로 재연결을 시도합니다.
 *
 * @example
 * ```ts
 * const ws = new ReconnectingWebSocket('wss://example.com/socket', {
 *   onMessage: (e) => console.log(e.data),
 *   onStatusChange: (status) => console.log('상태:', status),
 * });
 * ws.connect();
 * ```
 */
export class ReconnectingWebSocket {
  private readonly url: string;
  private readonly options: Required<
    Omit<ReconnectingWebSocketOptions, 'onStatusChange'>
  >;
  private readonly onStatusChange?: (status: WebSocketStatus) => void;
  private readonly handlers: WebSocketEventHandlers;

  private socket: WebSocket | null = null;
  private retryCount = 0;
  private retryTimer: ReturnType<typeof setTimeout> | null = null;
  private status: WebSocketStatus = 'disconnected';
  private isManuallyClosed = false;

  constructor(
    url: string,
    handlers: WebSocketEventHandlers = {},
    options: ReconnectingWebSocketOptions = {},
  ) {
    this.url = url;
    this.handlers = handlers;
    this.options = {
      ...DEFAULT_OPTIONS,
      initialDelay: options.initialDelay ?? DEFAULT_OPTIONS.initialDelay,
      maxDelay: options.maxDelay ?? DEFAULT_OPTIONS.maxDelay,
      maxRetries: options.maxRetries ?? DEFAULT_OPTIONS.maxRetries,
      backoffMultiplier:
        options.backoffMultiplier ?? DEFAULT_OPTIONS.backoffMultiplier,
    };
    this.onStatusChange = options.onStatusChange;
  }

  /** 현재 연결 상태를 반환합니다. */
  getStatus = (): WebSocketStatus => this.status;

  /** WebSocket 연결을 시작합니다. */
  connect = (): void => {
    this.isManuallyClosed = false;
    this.createSocket();
  };

  /**
   * WebSocket 연결을 수동으로 종료합니다.
   * 수동 종료 시 자동 재연결을 시도하지 않습니다.
   */
  disconnect = (): void => {
    this.isManuallyClosed = true;
    this.clearRetryTimer();

    if (this.socket) {
      this.socket.close();
      this.socket = null;
    }

    this.setStatus('disconnected');
  };

  /**
   * 메시지를 전송합니다.
   * @param data - 전송할 데이터
   */
  send = (data: string | ArrayBufferLike | Blob | ArrayBufferView): void => {
    if (this.socket?.readyState === WebSocket.OPEN) {
      this.socket.send(data);
    } else {
      console.warn('[ReconnectingWebSocket] 연결되지 않은 상태에서 send 호출됨');
    }
  };

  private createSocket = (): void => {
    if (this.socket) {
      this.socket.onopen = null;
      this.socket.onmessage = null;
      this.socket.onclose = null;
      this.socket.onerror = null;
      this.socket.close();
    }

    this.setStatus(this.retryCount > 0 ? 'reconnecting' : 'connecting');

    const socket = new WebSocket(this.url);

    socket.onopen = (event) => {
      this.retryCount = 0;
      this.setStatus('connected');
      this.handlers.onOpen?.(event);
    };

    socket.onmessage = (event) => {
      this.handlers.onMessage?.(event);
    };

    socket.onerror = (event) => {
      this.handlers.onError?.(event);
    };

    socket.onclose = (event) => {
      this.handlers.onClose?.(event);

      if (this.isManuallyClosed) {
        return;
      }

      this.scheduleReconnect();
    };

    this.socket = socket;
  };

  private scheduleReconnect = (): void => {
    const { maxRetries, initialDelay, maxDelay, backoffMultiplier } =
      this.options;

    const hasExceededRetries =
      maxRetries > 0 && this.retryCount >= maxRetries;

    if (hasExceededRetries) {
      console.error(
        `[ReconnectingWebSocket] 최대 재연결 시도 횟수(${maxRetries}회) 초과. 연결을 포기합니다.`,
      );
      this.setStatus('failed');
      return;
    }

    const delay = Math.min(
      initialDelay * Math.pow(backoffMultiplier, this.retryCount),
      maxDelay,
    );

    this.retryCount += 1;
    this.setStatus('reconnecting');

    console.info(
      `[ReconnectingWebSocket] ${delay}ms 후 재연결 시도 (${this.retryCount}/${maxRetries || '∞'})`,
    );

    this.retryTimer = setTimeout(() => {
      this.createSocket();
    }, delay);
  };

  private clearRetryTimer = (): void => {
    if (this.retryTimer !== null) {
      clearTimeout(this.retryTimer);
      this.retryTimer = null;
    }
  };

  private setStatus = (next: WebSocketStatus): void => {
    if (this.status === next) {
      return;
    }

    this.status = next;
    this.onStatusChange?.(next);
  };
}
```

## React 훅으로 감싸기 — `src/hooks/useReconnectingWebSocket.ts`

```typescript
import { useCallback, useEffect, useRef, useState } from 'react';

import {
  ReconnectingWebSocket,
  ReconnectingWebSocketOptions,
  WebSocketStatus,
} from '@/utils/ReconnectingWebSocket';

/** useReconnectingWebSocket 반환값 */
export interface UseReconnectingWebSocketReturn {
  status: WebSocketStatus;
  send: (data: string) => void;
  disconnect: () => void;
}

/**
 * 자동 재연결 WebSocket 훅.
 * 컴포넌트 언마운트 시 자동으로 연결을 해제합니다.
 *
 * @param url - WebSocket 서버 URL
 * @param onMessage - 메시지 수신 콜백
 * @param options - 재연결 옵션
 */
export const useReconnectingWebSocket = (
  url: string,
  onMessage: (event: MessageEvent) => void,
  options?: ReconnectingWebSocketOptions,
): UseReconnectingWebSocketReturn => {
  const [status, setStatus] = useState<WebSocketStatus>('disconnected');
  const wsRef = useRef<ReconnectingWebSocket | null>(null);
  const onMessageRef = useRef(onMessage);

  // 최신 핸들러 참조 유지 (리렌더링 없이)
  useEffect(() => {
    onMessageRef.current = onMessage;
  }, [onMessage]);

  useEffect(() => {
    const ws = new ReconnectingWebSocket(
      url,
      {
        onMessage: (event) => onMessageRef.current(event),
        onError: (event) => console.error('[WebSocket] 에러:', event),
      },
      {
        ...options,
        onStatusChange: setStatus,
      },
    );

    wsRef.current = ws;
    ws.connect();

    return () => {
      ws.disconnect();
    };
  }, [url]); // url 변경 시 재연결

  const send = useCallback((data: string) => {
    wsRef.current?.send(data);
  }, []);

  const disconnect = useCallback(() => {
    wsRef.current?.disconnect();
  }, []);

  return { status, send, disconnect };
};
```

## 사용 예시

```tsx
const ChatRoom = React.memo(() => {
  const { status, send } = useReconnectingWebSocket(
    'wss://api.example.com/chat',
    (event) => {
      console.log('수신:', event.data);
    },
    {
      initialDelay: 1000,
      maxDelay: 30000,
      maxRetries: 10,
    },
  );

  return (
    <div>
      <span>상태: {status}</span>
      <button onClick={() => send('hello')}>전송</button>
    </div>
  );
});
```

## 핵심 동작 정리

| 항목 | 내용 |
|------|------|
| **재연결 전략** | 지수 백오프 (1s → 2s → 4s → … → 최대 30s) |
| **수동 종료** | `disconnect()` 호출 시 재연결 없이 완전 종료 |
| **최대 시도** | `maxRetries: 0` 설정 시 무제한 재시도 |
| **상태 추적** | `connecting` → `connected` / `reconnecting` → `failed` |
| **메모리 누수 방지** | 이전 소켓 이벤트 핸들러 제거 후 새 소켓 생성 |
| **React 훅** | 언마운트 시 자동 정리, `onMessage` ref로 클로저 문제 방지 |
