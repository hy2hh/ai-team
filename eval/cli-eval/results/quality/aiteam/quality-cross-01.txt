#### Task 1-3: Webhook 엔드포인트 구현
- **담당**: Homer (Backend)
- **Files**: `app/api/webhooks/stripe/route.ts`
- **Steps**:
  - [ ] raw body 수신 설정
  - [ ] 서명 검증 로직

  - [ ] idempotency 처리 (중복 이벤트 스킵)
- **Code** (핵심):
```typescript
// app/api/webhooks/stripe/route.ts
import { headers } from 'next/headers';
import { stripe } from '@/lib/stripe';
import { prisma } from '@/lib/prisma';

export async function POST(req: Request) {
  const body = await req.text(); // raw body 필수
  const signature = headers().get('stripe-signature')!;

  let event: Stripe.Event;
  try {
    event = stripe.webhooks.constructEvent(
      body,
      signature,
      process.env.STRIPE_WEBHOOK_SECRET!,
    );
  } catch {
    return new Response('Invalid signature', { status: 400 });
  }

  // Idempotency 체크
  const already = await prisma.processedWebhookEvent.findUnique({
    where: { stripeEventId: event.id },
  });
  if (already) return new Response('Already processed', { status: 200 });

  await handleEvent(event);

  await prisma.processedWebhookEvent.create({
    data: { stripeEventId: event.id },
  });

  return new Response('OK', { status: 200 });
}
```

#### Task 2-2: Checkout 세션 API
- **담당**: Homer (Backend)
- **Files**: `app/api/checkout/session/route.ts`
- **Code**:
```typescript
export async function POST(req: Request) {
  const session = await getServerSession(); // 인증 확인
  if (!session?.user) return new Response('Unauthorized', { status: 401 });

  // Stripe Customer 조회 또는 생성
  let customer = await prisma.stripeCustomer.findUnique({
    where: { userId: session.user.id },
  });
  if (!customer) {
    const stripeCustomer = await stripe.customers.create({
      email: session.user.email!,
      metadata: { userId: session.user.id },
    });
    customer = await prisma.stripeCustomer.create({
      data: { userId: session.user.id, customerId: stripeCustomer.id },
    });
  }

  const checkoutSession = await stripe.checkout.sessions.create({
    customer: customer.customerId,
    mode: 'subscription',
    line_items: [{ price: process.env.STRIPE_PRICE_ID!, quantity: 1 }],
    success_url: `${process.env.NEXT_PUBLIC_APP_URL}/payment/success?session_id={CHECKOUT_SESSION_ID}`,
    cancel_url: `${process.env.NEXT_PUBLIC_APP_URL}/pricing`,
  });

  return Response.json({ url: checkoutSession.url });
}
```
